   # Оберіть файл
   for f in *.yml; do
       count=$(grep -o '\[[a-z_]*|e\]' "$f" 2>/dev/null | wc -l)
       if [ $count -gt 0 ] && [ $count -lt 100 ]; then
           echo "$count $f"
       fi
   done
   
   # Виправте його (3 команди)
   python3 ../../scripts/fix_any_file_universal.py filename.yml
   python3 ../../scripts/auto_translate_from_russian.py filename.yml
   python3 ../../scripts/fix_hints_with_icon.py
